\begin{frame}{多項式時間前処理の失われた大陸}

	\begin{block}{\textbf{素朴な定義} (有用な前処理アルゴリズム)}
		パラメータ化問題 \(Q\) の\alert{有用な前処理アルゴリズム}とは，任意のインスタンス \(I \in \Sigma^*\) を多項式時間で，同値なインスタンス \(I'\) であって \(|I'| < |I|\) を満たすものに変換するアルゴリズムである（と定義してみる）．
	\end{block}

	\begin{alertblock}{問題点}
		NP困難な問題に対してこのような前処理アルゴリズムが存在するならば，
		繰り返し適用することでインスタンスが定数サイズになり，
		$\mathcal{P} = \mathcal{NP}$ が導かれる．
	\end{alertblock}

	\note[item]{もし　\(\mathcal{P} \neq \mathcal{NP}\) を信じるならば，そのような前処理アルゴリズムは存在しないことになる．}
	\note[item]{問いの立て方が間違っていた．パラメータ \(k\) の観点から何か上手いことはいえないだろうか．}
\end{frame}

\begin{frame}{\PLC のカーネル}
	\begin{block}{Reduction PLC.}
		\(k+1\) 点を通る直線 \(\ell\) が存在するならば，\(\cP\) から \(\ell\) の点を削除し，\(k\) を \(1\) 減らす．
	\end{block}

	\structure{\textbf{Reduction PLC.} が適用できないインスタンス}

	\begin{claim}
		\((\cP, k)\) について \textnormal{\textbf{Reduction PLC.}} がこれ以上適用できないとする．
		このとき，\((\cP, k) \in\) \PLC ならば，\(|P| \le k^2\) である．
	\end{claim}

	\only<1>{
		\textbf{証明. }
		Yes インスタンスなので，\(|\cP|\) 点を高々 \(k\) 本の直線でカバーできる．また \textbf{Reduction PLC.} が適用できないことから，各直線は高々 \(k\) 点しかカバーしない．\hfill\qed
	}

	\only<2>{
		主張の対偶を取れば，\(|\cP| > k^2\) の場合，即座に No インスタンスと結論づけられる．
		\begin{theorem}[\cite{lokshtanov2009new}, \PLC のカーネルサイズ]
			\PLC 問題は \(k^2\) 点のカーネルを持つ．
		\end{theorem}
	}
\end{frame}

\begin{frame}{形式的な定義：還元規則，同値}
	\begin{definition}[還元規則, reduction rule]
		パラメータ化問題 \(Q\) の\alert{還元規則}とは，関数 \(\varphi \colon \Sigma^* \times \mathbb{N} \to \Sigma^* \times \mathbb{N}\) であり，\(Q\) のインスタンス \((I, k)\) を\dotuline{同値}なインスタンス \((I', k')\) に写像するもので，\(\varphi\) は \(|I|\) と \(k\) の多項式時間で計算可能であるもの．
	\end{definition}

	\begin{definition}[同値, equivelent]
		パラメータ化問題 \(Q\) のインスタンス \((I, k)\) と \((I', k')\) が\alert{同値}であるとは，\((I, k) \in Q\) ならば，またそのときに限り \((I', k') \in Q\) であること．
	\end{definition}

	\note[item]{ }

\end{frame}

\begin{frame}{形式的な定義：前処理アルゴリズム，出力サイズ，カーネル化アルゴリズム}
	パラメータ化問題 \(Q\) の\alert{前処理アルゴリズム (preprocessing algorithm)}とは，\((I, k)\) を多項式時間で同値な \((I', k')\) に変換するアルゴリズム．\onslide<2>{前処理アルゴリズム \(\mathcal{A}\) の\alert{出力サイズ (output size)} \(\text{size}_{\mathcal{A}} \colon \mathbb{N} \to \mathbb{N} \cup \{\infty\}\) とは，\[
			\text{size}_{\mathcal{A}}(k) = \sup \{ |I'| + k' : (I', k') = \mathcal{A}(I, k), I \in \Sigma^* \}
		\] で定義される関数．}

			ある計算可能な関数 \(g\) に対して

	}

\begin{frame}{例：\VC}

	\begin{problem}[\VC]
	グラフ \(G\) と整数 \(k\) が与えられたとき，\(S \subseteq V(G)\) であって \(|S| \le k\) かつ全ての辺の少なくとも一方の端点が \(S\) に属するものが存在するかどうかを判定せよ．
	\end{problem}

	\begin{block}{Reduction VC.1.}
		孤立点を削除する．
	\end{block}

	\begin{block}{Reduction VC.2.}
		次数 \(k+1\) 以上の頂点 \(v\) が存在するならば，\(v\) を削除し，\(k\) を \(1\) 減らす．
	\end{block}

	\begin{block}{Reduction VC.3.}
		\((G, k)\) を \textbf{Reduction VC.1, VC2} が適用できないインスタンスとする．
		このとき，\(k < 0\) ...
	\end{block}

	\begin{theorem}[\cite{buss1993nondeterminism}]
		\VC 問題は \(k^2 + k\) 頂点のカーネルを持つ．
	\end{theorem}

\end{frame}

\begin{frame}{カーネル化の定義と歴史的背景}
	\structure{初期の定義では \(k' < k\) が要求されていた（例えば \cite{niedermeier2006invitation} でみられる）．}

	\begin{definition}[以前のカーネル化アルゴリズム]
		パラメータ化問題 \(Q\) の\alert{カーネル化アルゴリズム}とは，\(Q\) の前処理アルゴリズムであり，ある計算可能な関数 \(g\) に対して \((I', k')\) がインスタンス \((I, k)\) に対する出力であるとき，
		\[
			\alert{k' < k} \quad \text{かつ} \quad |I'| \le g(k)
		\]
		が成り立つもの．
	\end{definition}

	\structure{現在では，\(k' < k\) という条件は緩和されている．}

	\note[item]{\(k\) が問題の複雑さを表すのなら，縮約によって \(k\) が厳密に小さくなることを求めるのは自然なような気もするが，，，}
	\note[item]{なお本書では，\(k' > k\) となるような前処理アルゴリズムは登場しない}

\end{frame}

	\begin{claim}
		パラメータ化問題 \(Q\) を有限時間で解く何らかのアルゴリズムが存在するとする．
		\(Q\) がカーネル化アルゴリズムを持つならば，\(Q\) は \FPT である．
	\end{claim}

	\textbf{Proof.} \(Q\) に対する \FPT アルゴリズムを構成する．

	インスタンス \((I, k)\) に対して，\(\text{poly}(|I|)\) 時間でカーネル \((I', k')\) を取り出す．
	その後，\(Q\) に対する何らかのアルゴリズムを実行する．

	\note[item]{例えば停止問題やヒルベルトの第 10 問題（整数係数の多項式方程式が整数解をもつかどうか判定）は，判定するアルゴリズムが存在しないため，カーネル化アルゴリズムがあろうとも \FPT であるとは言い切れない}

					\end{block}

	\begin{lemma}[Lemma 2.2, p.20]
		パラメータ化問題 \(Q\) について，もし \(Q\) が \FPT ならば，\(Q\) はカーネル化を持つ．
	\end{lemma}

	\textbf{Proof.}
	\(Q\) が \FPT なので，ある計算可能な関数 \(f\) と定数 \(c\) に対して，\\
	時間 \(f(k) \cdot |I|^c\) で \((I, k) \in Q\) かどうかを判定するアルゴリズム \(A\) が存在する．

	カーネル化アルゴリズムを構成する．
	入力 \((I, k)\) に対し，\(A\) を最大 \(|I|^{c+1}\) ステップだけ実行する．終了すれば yes/no の自明なインスタンスを返す．
	終了しなければ，\[
		f(k) \cdot |I|^{c} > |I|^{c+1} \implies |I| < f(k)
	\] であるから，\((I, k)\) をカーネルとして返す．\hfill\qed

	\note[item]{\(Q\) が空集合でも，その補集合でもない限り自明な yes/no のインスタンスは存在する．}
\end{frame}

\begin{frame}{\FPT とカーネル化の関係}

	\begin{theorem}[\cite{niedermeier2006invitation}]
		決定可能問題 \(Q\) について，\(Q\) は \FPT であることと \(Q\) がカーネル化アルゴリズムを持つことは同値である．
	\end{theorem}


	\begin{remark}
		ここで作られるカーネル化アルゴリズムは，多くの場合指数カーネルであり，あまり実用的でない
	\end{remark}
\end{frame}

\begin{frame}{まとめと今後の予定}

		\item パラメータ化問題が \FPT であることは，問題が決定可能でありカーネル化アルゴリズムを持つことと同値である
	\end{itemize}
